class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def change_name(self, new_name):
        self.name = new_name

class Student(Person):
    def __init__(self, name, age, university):
        super().__init__(name, age)
        self.university = university
        self.grades = []

    def add_grade(self, grade):
        self.grades.append(grade)

    def calculate_average_grade(self):
        if not self.grades:
            return 0
        return sum(self.grades) / len(self.grades)

student = Student("Дмитро", 19, "РФКІТ")

student.change_name("Діма")

student.add_grade(3)
student.add_grade(5)
student.add_grade(4)

average_grade = student.calculate_average_grade()

print("Ім'я:", student.name)
print("Вік:", student.age)
print("Університет:", student.university)
print("Середня оцінка:", average_grade)
